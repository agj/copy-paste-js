### uniq

```
var uniq = function uniq(list) {
  var seen = [];
  return list.filter(function (item) {
    return seen.some(function (a) {
      return a === item;
    }) ? false : (seen.push(item), true);
  });

### throttle

```
var throttle = function throttle(secs, fn) {
  var waitTime = secs * 1000;
  var last = 0;
  return function () {
    var now = Date.now();

      last = now;
      fn.apply(void 0, arguments);
    }
  };
